<script>{`
  // Aggressive navigation hiding (excluding search bar - handled by CSS)
  function hideNavElements() {
    const selectors = [
      '.nav-tabs',
      'div.nav-tabs',
      '.nav-tabs-item',
      'a.nav-tabs-item',
      '.link.nav-tabs-item',
      'button[aria-haspopup="menu"]',
      'button[data-state]',
      'button.group',
      'button[class*="bg-background-light"]',
      'button[class*="bg-background-dark"]',
      'header button:not([aria-label*="theme"]):not([aria-label*="Theme"]):not([id="search-bar-entry"])',
      'header a.link',
      'header .link',
      '[class*="rounded-full"]:not([aria-label*="theme"]):not([aria-label*="Theme"])',
      '[class*="lg:flex"]',
      'nav',
      '[role="navigation"]',
      'header a:not([href="/"])',
      '[class*="version"]',
      '[class*="Version"]',
      '[class*="breadcrumb"]'
    ];

    selectors.forEach(selector => {
      try {
        document.querySelectorAll(selector).forEach(el => {
          el.style.setProperty('display', 'none', 'important');
          el.style.setProperty('visibility', 'hidden', 'important');
          el.style.setProperty('opacity', '0', 'important');
          el.style.setProperty('pointer-events', 'none', 'important');
          el.style.setProperty('position', 'absolute', 'important');
          el.style.setProperty('left', '-9999px', 'important');
        });
      } catch(e) {}
    });
  }

  // Theme detection
  function applyTheme() {
    const html = document.documentElement;
    const theme = html.getAttribute('data-theme') ||
                  (html.classList.contains('dark') ? 'dark' : 'light');
    const customContent = document.getElementById('custom-index-content');
    if (customContent) {
      customContent.setAttribute('data-theme', theme);
    }
  }

  // Run immediately and repeatedly
  if (typeof window !== 'undefined') {
    hideNavElements();
    setTimeout(hideNavElements, 0);
    setTimeout(hideNavElements, 100);
    setTimeout(hideNavElements, 500);
    setTimeout(hideNavElements, 1000);

    window.addEventListener('load', hideNavElements);
    document.addEventListener('DOMContentLoaded', hideNavElements);

    // Apply theme on load
    applyTheme();
    setTimeout(applyTheme, 0);
    setTimeout(applyTheme, 100);
    setTimeout(applyTheme, 500);
    window.addEventListener('load', applyTheme);
    document.addEventListener('DOMContentLoaded', applyTheme);

    // Watch for dynamic content and theme changes
    const observer = new MutationObserver(() => {
      hideNavElements();
      applyTheme();
    });

    // Watch HTML element for theme changes
    observer.observe(document.documentElement, {
      attributes: true,
      attributeFilter: ['data-theme', 'class']
    });

    if (document.body) {
      observer.observe(document.body, {
        childList: true,
        subtree: true,
        attributes: true,
        attributeFilter: ['class', 'style']
      });
    } else {
      setTimeout(() => {
        if (document.body) {
          observer.observe(document.body, {
            childList: true,
            subtree: true,
            attributes: true,
            attributeFilter: ['class', 'style']
          });
        }
      }, 100);
    }
  }
`}</script>
