{/* JavaScript to hide navigation after page loads */}
<script dangerouslySetInnerHTML={{
  __html: `
    function hideNavigation() {
      const selectors = [
        // Navigation tabs
        '[data-testid="navigation-tabs"]',
        '[data-testid*="tab"]',
        '.navigation-tabs',
        'nav[role="navigation"]',
        'div[role="tablist"]',
        '[role="tab"]',
        '[role="tabpanel"]',
        // Version switcher
        '[data-testid="version-picker"]',
        '.version-picker',
        '.hidden.lg\\:flex.items-center.gap-x-2',
        '[class~="hidden"][class~="lg:flex"][class~="items-center"][class~="gap-x-2"]',
        // Header navigation
        'header nav',
        '.mint-nav'
      ];

      // Also try to find elements by their text content
      const tabTexts = ['Documentation', 'API Reference', 'Release'];
      document.querySelectorAll('*').forEach(el => {
        if (tabTexts.includes(el.textContent?.trim()) &&
            el.tagName !== 'HTML' && el.tagName !== 'BODY' &&
            el.offsetWidth > 0 && el.offsetHeight > 0) {
          // Hide the element and its parent if it looks like a tab
          if (el.parentElement && (
              el.parentElement.role === 'tablist' ||
              el.parentElement.className.includes('tab') ||
              el.parentElement.className.includes('nav')
          )) {
            el.parentElement.style.display = 'none';
          } else {
            el.style.display = 'none';
          }
        }
      });

      selectors.forEach(selector => {
        try {
          const elements = document.querySelectorAll(selector);
          elements.forEach(el => {
            el.style.display = 'none';
            el.style.visibility = 'hidden';
            el.style.height = '0';
            el.style.overflow = 'hidden';
          });
        } catch(e) {
          console.log('Selector failed:', selector, e);
        }
      });

    }

    // Run immediately and also after DOM loads
    hideNavigation();
    document.addEventListener('DOMContentLoaded', hideNavigation);
    window.addEventListener('load', hideNavigation);

    // Also run on any changes (for dynamic content)
    setTimeout(hideNavigation, 100);
    setTimeout(hideNavigation, 500);
    setTimeout(hideNavigation, 1000);
  `
}} />
